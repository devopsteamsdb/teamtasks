1. Get the application URL by running these commands:
{{- if .Values.route.enabled }}
  export ROUTE_HOST=$(oc get route {{ include "teamtasks.fullname" . }} -o jsonpath='{.spec.host}')
  echo "TeamTasks is available at: https://$ROUTE_HOST"
{{- else if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(oc get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "teamtasks.fullname" . }})
  export NODE_IP=$(oc get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'oc get --namespace {{ .Release.Namespace }} svc -w {{ include "teamtasks.fullname" . }}'
  export SERVICE_IP=$(oc get svc --namespace {{ .Release.Namespace }} {{ include "teamtasks.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(oc get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "teamtasks.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(oc get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  oc --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Check the status of the deployment:
  oc get pods -l "app.kubernetes.io/name={{ include "teamtasks.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. View logs:
  oc logs -f deployment/{{ include "teamtasks.fullname" . }}

4. Access the admin panel:
  Open the application URL and navigate to /admin

5. Persistent data locations:
  - Database: /app/instance (PVC: {{ include "teamtasks.fullname" . }}-database)
  - Uploads: /app/uploads (PVC: {{ include "teamtasks.fullname" . }}-uploads)
